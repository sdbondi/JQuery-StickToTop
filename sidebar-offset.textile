  var $stickyNav = $('.sticky'),
      navHeight	 = $stickyNav.outerHeight(true);
  $stickyNav.stickToTop();
  $('.sticky-sidebar').stickToTop({ offset: {top: navHeight}});
group: examples
---

For situations where you need to offset where an element begins to stick, such as when sticking a sidebar and a navigation simultaneously, the @offset@ option can be used:


{% highlight javascript %}
// get the height of the navigation
var navHeight = $('.sticky').outerHeight(true);

$('.sticky-sidebar').stickToTop({ offset: {top: navHeight}});
{% endhighlight %}
